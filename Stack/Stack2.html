CREATE A STACK IMPLEMENTING THE OPERATIONS ON THE CHARACTERS ARRAY (PUSH, POP, PEEK, ISEMPTY, ISFULL)

#include <stdio.h>
      
int push(char[],int,char);

int pop(char[],int);
int peek(char[],int);

void main()
{
    char stack[100],val;
    int ch,i=-1;
    do
    {
        printf("Enter Choise\n");
        scanf("%d",&ch);
        switch (ch)
        {
        case 1:
            printf("Enter Character\n");
            scanf(" %c",&val);
            i=push(stack,i,val);
            break;
            case 2:
            i=pop(stack,i);
            break;
            case 3:
            i=peek(stack,i);
            if (i==-1)
            {
                printf("Stack is Empty\n");
            }
            else
            {
                printf("Topest character is %c",stack[i]);
            }
            break;
        default:
        printf("erroe\n");
            break;
        }
    } while (ch!=4);
    
}

int push(char stack[],int i,char op)
{
    if(i==100)
    {
    printf("Stack is full\n");
    return -1;
    }
    else{
        i++;
        stack[i]=op;
        printf("%c\n",stack[i]);
        //printf("push\n");
        return i;
    }
}
int pop(char stack[],int i)
{
    if(i==-1)
    {
        printf("Stack is empty\n");
        return -1;
    }
    else
    {
        i--;
        return i;
    }
}
int peek(char stack[],int i)
{
    if(i==-1)
    return -1;
    else
    return i;
}
